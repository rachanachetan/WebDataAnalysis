# Calculate the Within Clusters Sum of Squares
wcss = []
for i in range(1,10):
    kmeans= KMeans(n_clusters = i, init = 'k-means++', random_state = 42)
    kmeans.fit(online_std)
    wcss.append(kmeans.inertia_)
In [ ]:
